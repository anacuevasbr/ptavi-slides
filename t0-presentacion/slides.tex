\section{Filosofía y Evaluación}


\begin{frame}
\frametitle{Evaluación}

\begin{itemize}
\item Prácticas: hasta 6 puntos. \\ Mínimo: Parte básica práctica final.
\item Teoría: hasta 5 puntos. \\ Mínimo: 2 puntos.
\item Al sumar teoría y prácticas, la suma ha de dar un 5 o más.
\item Además, se puede conseguir medio puntito adicional (ver más adelante).
\item Novedad: Las tareas serán individuales, pero podrás obtener puntuación extra por el desempeño de tu equipo.
\end{itemize}


\end{frame}

\begin{frame}
\frametitle{Más cosas sobre la evaluación}

\begin{itemize}
  \item Convocatorias
  \begin{itemize}
    \item Si uno aprueba teoría o a prácticas en diciembre, se ``guarda'' su nota para junio.
    \item No se ``guarda'' nota de un curso para otro.
  \end{itemize}
  \item Matrículas de honor: Algoritmo
  \begin{itemize}
    \item Si sumas más de diez, obtienes un diez
    \item En caso de que haya más alumnos con uno superior a diez que matrículas de honor, se darán a aquéllos con la suma de notas más alta
    \item Al resto, les regalaremos un libro (dedicado, si quieres)
  \end{itemize}
\end{itemize}

\end{frame}


\begin{frame}
\frametitle{Evaluación Prácticas (I)}

\begin{itemize}
    \item Han ser ser realizadas de manera individual
\end{itemize}

\begin{itemize}
\item Evaluación continua (hasta 3,5 puntos)
  \begin{itemize}
    \item Las prácticas marcadas con un (*) se entregarán el lunes 
          siguiente (hasta las 23:55) y darán hasta 0,7 puntos cada una.
  \end{itemize}
\item Práctica final (hasta 2 puntos)
  \begin{itemize}
    \item Requisitos mínimos ({\bf obligatorios}) y adicionales (opcionales)
    \item Los requisitos mínimos puede que no sean iguales para todos
    \item Enunciado: Mediados de noviembre. Entrega: Día de examen
    \item Habrá clases específicas de apoyo a la práctica final
  \end{itemize}
\item Según cómo desempeñe tu equipo, podrás conseguir hasta 0,5 puntos más.
\end{itemize}

\end{frame}


\begin{frame}
\frametitle{Evaluación Prácticas (y II)}

\begin{itemize}
\item Puntuación máxima:
  \begin{itemize}
    \item Entrega 1 - 29.09 (``Python y GitHub''): 0,7 puntos
    \item Entrega 2 - 13.10 (``SMIL, XML y Python''): 0,7 puntos
    \item Entrega 3 - 27.10 (``Sockets, UDP y Python''): 0,7 puntos
	\item Entrega 4 - 10.11 (``Sesión SIP + RTP''): 0,7 puntos
    \item Entrega 5 - 24.11 (``Registrar simple''): 0,7 puntos
    \item Práctica final - parte básica: 0 puntos
    \item Práctica final - parte adicional: 2 puntos
    \item Desempeño equipo: 0,5 puntos
  \end{itemize}
\end{itemize}

\end{frame}




\begin{frame}
\frametitle{Evaluación Teoría}


\begin{itemize}
\item {\bf Evaluación final global de teoría}:
\begin{itemize}
  \item Fecha: diciembre (convocatoria de diciembre)
  \item Hasta 2,5 puntos (mínimo 1 punto)
\end{itemize}
\item {\bf Método PEPEOLA}:
  \begin{itemize}
    \item Hasta 2,5 puntos (mínimo 1 punto)
    \item Se ha de venir con el tema del día preparado
    \item Una semana antes se abrirá el tema en el Moodle y se indicarán lecturas (transparencias, documentos, enlaces) y/o vídeos
    \item Habrá tests de evaluación (limitado en el tiempo: 10 minutos para responder) para cada clase que habrá que realizar ANTES de clase.
    \item ``As en la manga'': el 10.11 podemos \emph{declarar nulo} el método PEPEOLA.
  \end{itemize}
\end{itemize}

\end{frame}

\begin{itemize}

  %\item Los alumnos han de ver que este método vale la pena.
  \item Tras hacer los pertinentes cálculos, el tiempo de preparación
por clase se ha fijado en una hora y media (incluido el test):
  \begin{itemize}
    \item El primer test será de prueba para que los alumnos se familiaricen
    \item Quedan 10 temas de teoría, para los cuales habrá 5 tests
    \item Hay que superar 4 de los 5 tests
    \item Se supera un tema con 6 o más respuestas acertadas (de 10)
    \item Aplicamos una función ``rara'' para calcular la nota final
    \item Habrá un plus por desempeño positivo del equipo
  \end{itemize}
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Moodle}

\begin{itemize}
\item Se ruega poner foto vuestra y datos verídicos en vuestra ficha.
\item Se pedirá el tiempo que te ha llevado prepararte cada clase (pero eso no se evalúa, es un dato estadístico).
\item A veces al mejor test de la semana, le damos un regalito. El mejor 10 es:
\begin{itemize}
  \item El que ha acertado todas las preguntas (antes de revisión)
  \item El que tiene foto (suya) en su perfil de Moodle
  \item En caso de haber más de uno, el que lo ha hecho en el menor tiempo
\end{itemize}
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Estimación de dedicación}

\begin{itemize}
\item Esta asignatura tiene 6 créditos ECTS
% (60 horas de clase presencial)
%\item Según el cálculo que se realiza para ECTS, 6 créditos de los ``antiguos''
%se transforman en unos 4,5 ECTS.
\item Cada ECTS supone una 
dedicación de un estudiante ``medio'' de 25 horas por crédito.
\item Salen {\bf 150 horas} para la asignatura.
\item Presencialidad (60 horas):
\begin{itemize}
  \item 12 clases de teoría (+1 de presentación)
  \item 12 clases de prácticas 
\end{itemize}
\item No presencialidad:
\begin{itemize}
  \item Preparación clase de teoría: 2 horas * clase de teoría
  \item Terminar prácticas a entregar: 6 horas * práctica
  \item Extra práctica final: 20 horas (8 de tutoría)
  \item Entrega práctica final: 1 hora
  \item Preparación evaluación: 5 horas
  \item Evaluación global final (teoría): 1 hora
  \item Semana que viene: 8 horas
\end{itemize}
\item Total = {\bf 133 horas}
\end{itemize}
